1//! This module provides the functionality for defining and managing log rules.
2//!
3//! A log rule consists of a matcher and an action. The matcher determines if a given log line should trigger
4//! the associated action. Log rules are stored and processed in order of their ranking.
5
6use crate::constants::INSTANCE_TARGET;
7use std::sync::{Arc, RwLock};
8use tracing::{error, info, trace, warn};
9
10/// The default ranking value for log rules.
11pub static DEFAULT_STOP_INT: i32 = 99999;
12
13/// A matcher function that determines whether a log line should trigger a rule.
14/// It takes a string slice and returns a boolean.
15pub type Matcher = Arc<dyn Fn(&str) -> bool + Send + Sync>;
16
17/// An action function that is executed when its corresponding matcher returns true.
18/// It takes a string slice as input.
19pub type Action = Arc<dyn Fn(&str) + Send + Sync>;
20
21/// Computes the default ranking for a new rule based on the current count of rules.
22fn default_ranking(current_count: usize) -> i32 {
23    trace!("Computing default ranking for rule count: {current_count}");
24    current_count as i32 - DEFAULT_STOP_INT
25}
26
27#[derive(Clone)]
28pub struct LogRule {
29    pub matcher: Matcher,
30    pub action: Action,
31    pub ranking: i32,
32    pub stop: bool,
33}
34
35impl Default for LogRule {
36    fn default() -> Self {
37        trace!("Creating default LogRule");
38        LogRule {
39            matcher: Arc::new(|_| true),
40            action: Arc::new(|line| info!(target: INSTANCE_TARGET, "{line}")),
41            ranking: DEFAULT_STOP_INT,
42            stop: true,
43        }
44    }
45}
46
47impl LogRule {
48    pub fn new() -> Self {
49        trace!("Creating new LogRule");
50        Self::default()
51    }
52}
53
54#[derive(Clone)]
55pub struct LogRules {
56    rules: Arc<RwLock<Vec<LogRule>>>,
57}
58
59impl LogRules {
60    pub fn new() -> Self {
61        trace!("Initializing LogRules");
62        Self {
63            rules: Arc::new(RwLock::new(vec![LogRule::default()])),
64        }
65    }
66
67    pub fn add_rule<F, G>(&self, matcher: F, action: G, stop: bool, ranking: Option<i32>)
68    where
69        F: Fn(&str) -> bool + Send + Sync + 'static,
70        G: Fn(&str) + Send + Sync + 'static,
71    {
72        trace!("Adding new rule with stop flag: {stop}");
73        let mut rules = self.rules.write().unwrap();
74        let mut rule = LogRule::new();
75        rule.stop = stop;
76        rule.matcher = Arc::new(matcher);
77        rule.action = Arc::new(action);
78        rule.ranking = ranking.unwrap_or_else(|| default_ranking(rules.len()));
79        rules.push(rule);
80    }
81
82    pub fn get_rules(&self) -> Vec<LogRule> {
83        trace!("Retrieving and sorting rules");
84        let mut rules = self.rules.read().unwrap().clone();
85        rules.sort_by_key(|r| r.ranking);
86        trace!("Sorted rules count: {}", rules.len());
87        rules
88    }
89}
90
91impl Default for LogRules {
92    fn default() -> Self {
93        trace!("Creating default LogRules instance");
94        let rules = Self::new();
95        rules.add_rule(
96            |line| line.contains("WARNING"),
97            |line| warn!(target: INSTANCE_TARGET, "{}", line),
98            true,
99            None,
100        );
101        rules.add_rule(
102            |line| line.contains("ERROR"),
103            |line| error!(target: INSTANCE_TARGET, "{}", line),
104            true,
105            None,
106        );
107        rules
108    }
109}
