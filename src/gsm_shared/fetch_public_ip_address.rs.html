1use reqwest::blocking::Client;
2use std::env::VarError;
3use std::{env, fmt};
4use tracing::{debug, error};

6#[derive(Debug, serde::Deserialize, serde::Serialize)]
7struct IPResponse {
8    ip: String,
9}
10
11pub struct IPConfig {
12    pub(crate) ip: String,
13    pub(crate) port: u16,
14}
15
16impl fmt::Display for IPConfig {
17    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
18        write!(f, "{}:{}", self.ip, self.port)
19    }
20}
21
22impl IPConfig {
23    fn new(ip: String, port: u16) -> IPConfig {
24        IPConfig { ip, port }
25    }
26
27    fn default() -> IPConfig {
28        IPConfig::new("127.0.0.1".to_string(), 2456)
29    }
30
31    fn get_ip_from_env(&self) -> Result<String, VarError> {
32        env::var("ADDRESS")
33    }
34
35    fn get_port_from_env(&self) -> Result<u16, VarError> {
36        env::var("PORT").map(|port| port.parse().unwrap())
37    }
38
39    pub fn to_string_from_env(&self) -> Result<IPConfig, VarError> {
40        match self.get_ip_from_env() {
41            Ok(ip) => match self.get_port_from_env() {
42                Ok(port) => {
43                    if ip.is_empty() {
44                        error!("IP address is empty");
45                        Err(VarError::NotPresent)
46                    } else if port.to_string().is_empty() {
47                        error!("Port is empty");
48                        Err(VarError::NotPresent)
49                    } else {
50                        Ok(IPConfig::new(ip, port))
51                    }
52                }
53                Err(e) => Err(e),
54            },
55            Err(e) => Err(e),
56        }
57    }
58
59    pub fn fetch_ip_from_api(&self, client: &Client) -> Result<String, Box<dyn std::error::Error>> {
60        let urls = [
61            "https://api.ipify.org?format=json",
62            "https://api.seeip.org/jsonip?",
63            "https://ipinfo.io",
64        ];
65
66        for url in urls {
67            match client.get(url).send() {
68                Ok(response) => match response.json::<IPResponse>() {
69                    Ok(json) => return Ok(json.ip.to_string()),
70                    Err(e) => {
71                        debug!("Failed to parse JSON from {}: {}", url, e);
72                        continue;
73                    }
74                },
75                Err(e) => {
76                    debug!("Request to {} failed: {}", url, e);
77                    continue;
78                }
79            }
80        }
81
82        Err(Box::new(std::io::Error::new(
83            std::io::ErrorKind::NotFound,
84            "All IP fetch attempts failed",
85        )))
86    }
87}

89// Standardized way of fetching public address.
90pub fn fetch_public_address() -> IPConfig {
91    let client = Client::new();
92    let mut ip_config = IPConfig::default();
93    debug!("Checking for address in env");
94    match ip_config.to_string_from_env() {
95        Ok(ip) => {
96            debug!("Fetched IP from env: {}", ip);
97            ip
98        }
99        Err(_) => match ip_config.fetch_ip_from_api(&client) {
100            Ok(ip) => {
101                debug!("Fetched IP from API: {}", ip);
102                ip_config.ip = ip;
103                ip_config
104            }
105            Err(e) => {
106                debug!("Failed to fetch IP from API: {}", e);
107                ip_config
108            }
109        },
110    }
111}

118    #[test]
119    fn test_to_string_from_env_success() {
120        let key_address = "ADDRESS";
121        let key_port = "PORT";
122        let expected_address = "192.168.1.100";
123        let expected_port = "8080";
124        // Unsafe blocks required for setting env vars in Rust 2024.
125        unsafe {
126            env::set_var(key_address, expected_address);
127            env::set_var(key_port, expected_port);
128        }
129        let ip_config = IPConfig::default();
130        let config = ip_config.to_string_from_env().unwrap();
131        assert_eq!(config.ip, expected_address);
132        assert_eq!(config.port, expected_port.parse::<u16>().unwrap());
133        unsafe {
134            env::remove_var(key_address);
135            env::remove_var(key_port);
136        }
137    }
