30pub fn path_exists(path: &str) -> bool {
31    let state = Path::new(path).exists();
32    debug!(
33        "Path {} {}",
34        path,
35        if state { "exists" } else { "does not exist" }
36    );
37    state
38}

40pub fn parse_file_name(url: &Url, default: &str) -> String {
41    String::from(
42        url.path_segments()
43            .and_then(|mut segments| segments.next_back())
44            .and_then(|name| if name.is_empty() { None } else { Some(name) })
45            .unwrap_or(default),
46    )
47}
