449        // If a PRESET env variable is provided, override our base.

452        {
453            settings.apply_preset(preset);

756/// Saves the configuration to an INI file.
757pub fn save_config(path: &Path, settings: &Settings) {
758    let ini_config = to_string(&settings).unwrap();
759
760    if let Err(e) = fs::write(path, ini_config) {
761        eprintln!("Failed to save config: {e}");
762    }
763}
764
765/// Loads the configuration from an INI file or returns defaults if the file is missing.
766pub fn load_or_create_config(path: &Path) -> GameSettings {
767    if !path.parent().unwrap().exists() {
768        create_dir_all(path.parent().unwrap()).unwrap();
769    }
770    let default_config = Settings::default();
771    save_config(path, &default_config);
772    default_config.option_settings
773}
