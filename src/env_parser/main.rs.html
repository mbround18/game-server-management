1use clap::Parser;
2use regex::Regex;
3use serde::{Deserialize, Serialize};
4use std::collections::HashMap;
5use std::error::Error;

7use std::path::{Path, PathBuf};

9#[derive(Parser, Debug)]
10#[clap(about = "Scans Rust projects for environment variables")]
11struct Cli {
12    /// Directory to scan for Cargo.toml files
13    directory: String,
14
15    /// Optional output file path
16    #[clap(long)]
17    output: Option<PathBuf>,
18}
19
20#[derive(Serialize, Deserialize, Debug, Default)]
21struct EnvVarInfo {
22    field: Option<String>,
23    var_type: Option<String>,
24    default: Option<String>,
25    description: String,
26}
27
28fn main() -> Result<(), Box<dyn Error>> {
29    let args = Cli::parse();
30    let cargo_files = find_cargo_toml_files(&args.directory)?;
31    println!("Found {} Cargo.toml files", cargo_files.len());
32
33    for cargo_file in cargo_files {
34        process_cargo_toml(&cargo_file, args.output.as_deref())?;
35    }
36
37    println!("Processing complete!");
38    Ok(())
39}

41fn find_cargo_toml_files(dir: &str) -> Result<Vec<PathBuf>, Box<dyn Error>> {
42    let mut result = Vec::new();
43    for entry in fs::read_dir(dir)? {
44        let path = entry?.path();
45        if path.is_dir() {
46            result.extend(find_cargo_toml_files(path.to_str().unwrap())?);
47        } else if path.file_name().is_some_and(|f| f == "Cargo.toml") {
48            result.push(path);
49        }
50    }
51    Ok(result)
52}
53
54fn process_cargo_toml(cargo_path: &Path, output_path: Option<&Path>) -> Result<(), Box<dyn Error>> {
55    println!("Processing: {}", cargo_path.display());
56    let project_dir = cargo_path.parent().ok_or("Missing parent directory")?;
57    let rust_files = find_rust_files(project_dir)?;
58    let mut env_vars: HashMap<String, EnvVarInfo> = HashMap::new();
59
60    for file in rust_files {
61        extract_env_vars_from_file(&file, &mut env_vars)?;
62    }
63
64    let out_path =
65        output_path.map_or_else(|| project_dir.join("variables.json"), |p| p.to_path_buf());
66    let json = serde_json::to_string_pretty(&env_vars)?;
67    fs::write(&out_path, json)?;
68    println!(
69        "Wrote {} variables to {}",
70        env_vars.len(),
71        out_path.display()
72    );
73    Ok(())
74}

76fn find_rust_files(dir: &Path) -> Result<Vec<PathBuf>, Box<dyn Error>> {
77    let mut rust_files = Vec::new();
78    for entry in fs::read_dir(dir)? {
79        let path = entry?.path();
80        if path.is_dir() {
81            rust_files.extend(find_rust_files(&path)?);
82        } else if path.extension().is_some_and(|e| e == "rs") {
83            rust_files.push(path);
84        }
85    }
86    Ok(rust_files)
87}

89fn extract_env_vars_from_file(
90    file_path: &Path,
91    env_vars: &mut HashMap<String, EnvVarInfo>,
92) -> Result<(), Box<dyn Error>> {
93    let content = fs::read_to_string(file_path)?;
94
95    // Each pattern is now in a raw string literal (r#"..."#),
96    // which avoids having to escape backslashes multiple times.
97    // We also allow multiline with (?s) and an optional trailing comma with (?:,)?
98    let patterns = vec![
99        (
100            r#"(?s)std::env::var\("([A-Z0-9_]+)"\)(?:,)?"#,
101            Some((1, None, None)),
102            None,
103        ),
104        (
105            r#"(?s)env::var\("([A-Z0-9_]+)"\)(?:,)?"#,
106            Some((1, None, None)),
107            None,
108        ),
109        (
110            r#"(?s)env::var\("([A-Z0-9_]+)"\)\.unwrap_or_else\(\|[_a-zA-Z]*\|\s*settings\.([a-zA-Z0-9_]+)\.clone\(\)\)(?:,)?"#,
111            Some((1, Some(2), None)),
112            None,
113        ),
114        (
115            // More flexible approach with optional whitespace after 'env_parse!' and optional trailing comma.
116            r#"(?s)env_parse!\s*\(\s*\"([A-Z0-9_]+)\"\s*,\s*(.*?)\s*,\s*([a-zA-Z0-9_:<>]+)\s*\)(?:,)?"#,
117            Some((1, None, Some(3))),
118            None,
119        ),
120        (
121            r#"(?s)std::env::var\("([A-Z0-9_]+)"\)\.unwrap_or_else\(\|[_a-zA-Z]*\|\s*"([^"]+)"\.to_string\(\)\)(?:,)?"#,
122            Some((1, None, None)),
123            None,
124        ),
125        (
126            r#"(?s)fetch_var\("([A-Z0-9_]+)"(?:,\s*"([^"]*)")?\)(?:,)?"#,
127            Some((1, None, None)),
128            Some("fetch_var"),
129        ),
130        (
131            r#"(?s)is_env_var_truthy\("([A-Z0-9_]+)"\)(?:,)?"#,
132            Some((1, None, None)),
133            Some("bool"),
134        ),
135    ];
136
137    for (pattern, groups, extra_type) in patterns {
138        let regex = Regex::new(pattern)?;
139        for caps in regex.captures_iter(&content) {
140            let var_name = caps.get(1).unwrap().as_str().to_string();
141            let entry = env_vars.entry(var_name.clone()).or_default();
142
143            // groups => (0: entire match) (1: var_name) (field_idx, type_idx)
144            if let Some((_, field_idx, type_idx)) = groups {
145                // If we have an index for the field, fill it
146                if let Some(f_idx) = field_idx
147                    && let Some(field_cap) = caps.get(f_idx)
148                {
149                    entry.field = Some(field_cap.as_str().to_string());
150                }
151                // If we have an index for the type, fill it
152                if let Some(t_idx) = type_idx
153                    && let Some(var_type_cap) = caps.get(t_idx)
154                {
155                    entry.var_type = Some(var_type_cap.as_str().to_string());
156                }
157                // fetch_var default is group(2)
158                if pattern.contains("fetch_var") && caps.get(2).is_some() {
159                    entry.default = Some(caps.get(2).unwrap().as_str().to_string());
160                }
161                // env_parse! default is group(2)
162                if pattern.contains("env_parse!") && caps.get(2).is_some() {
163                    entry.default = Some(caps.get(2).unwrap().as_str().trim().to_string());
164                }
165            }
166
167            // If "bool", we set the var_type
168            if let Some(hardcoded_type) = extra_type
169                && hardcoded_type == "bool"
170            {
171                entry.var_type = Some("bool".to_string());
172            }
173        }
174    }
175
176    Ok(())
177}
