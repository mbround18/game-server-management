4/// Applies environment variable overrides to the config.
5pub fn apply_env_overrides(config: &mut ServerConfig) {
6    let env_config = crate::game_settings::GameSettings::from_env();
7    config.game_settings.merge_env(&env_config);
8
9    for (key, value) in env::vars() {
10        if let Some(stripped) = key.strip_prefix("SET_GROUP_") {
11            let mut parts = stripped.splitn(2, '_');
12            if let (Some(group_name), Some(field_name)) = (parts.next(), parts.next())
13                && let Some(group) = config
14                    .user_groups
15                    .iter_mut()
16                    .find(|g| g.name.eq_ignore_ascii_case(group_name))
17            {
18                match field_name.to_lowercase().as_str() {
19                    "password" => group.password = value,
20                    "can_kick_ban" => {
21                        group.can_kick_ban = value.parse().unwrap_or(group.can_kick_ban)
22                    }
23                    "can_access_inventories" => {
24                        group.can_access_inventories =
25                            value.parse().unwrap_or(group.can_access_inventories)
26                    }
27                    _ => {}
28                }
29            }
30        }
31    }
32}

43    fn make_config_with_group(name: &str) -> ServerConfig {
44        ServerConfig {
45            user_groups: vec![UserGroup {
46                name: name.to_string(),
47                password: "oldpass".to_string(),
48                can_kick_ban: false,
49                can_access_inventories: false,
50                can_edit_base: false,
51                can_extend_base: false,
52                reserved_slots: 0,
53            }],
54            ..Default::default()
55        }
56    }
57
58    fn clear_env_var(var: &str) {
59        unsafe {
60            std::env::remove_var(var);
61        }
62    }
63
64    fn apply_env_var(var: &str, value: &str) {
65        unsafe {
66            std::env::set_var(var, value);
67        }
68    }

70    #[test]
71    fn test_password_override() {
72        let _lock = TEST_MUTEX.lock().unwrap();
73        let group = "Admin";
74        let env_var = format!("SET_GROUP_{group}_PASSWORD");
75        apply_env_var(&env_var, "newpass");
76        let mut config = make_config_with_group(group);
77        apply_env_overrides(&mut config);
78        assert_eq!(config.user_groups[0].password, "newpass");
79        clear_env_var(&env_var);
80    }
