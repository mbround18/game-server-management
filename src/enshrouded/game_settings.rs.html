123macro_rules! env_field_mapping {
124    ($($field:ident => $env_var:literal),*) => {
125        pub fn from_env() -> Self {
126            Self {
127                $(
128                    $field: env_parse!($env_var, Self::default().$field, _),
129                )*
130            }
131        }
132
133        pub fn merge_env(&mut self, env_config: &GameSettings) {
134            $(
135                if std::env::var($env_var).is_ok() {
136                    self.$field = env_config.$field.clone();
137                }
138            )*
139        }
140    };
141}

277/// Loads the configuration from a file or creates a new one with defaults.
278/// Environment variables override both file values and defaults.
279pub fn load_or_create_config(path: &Path) -> ServerConfig {
280    tracing::debug!("Loading config from path: {:?}", path);
281
282    let mut config = load_config_with_defaults::<ServerConfig>(path);
283    tracing::debug!("Config loaded: {:?}", config.game_settings);
284
285    let original_config = config.clone();
286
287    tracing::debug!("Config loaded, applying environment overrides");
288    apply_env_overrides(&mut config);
289
290    let config_changed =
291        serde_json::to_string(&config).unwrap() != serde_json::to_string(&original_config).unwrap();
292    tracing::debug!("Config changed after env overrides: {}", config_changed);
293
294    if path.exists() && config_changed {
295        let timestamp = chrono::Local::now().format("%Y-%m-%d-%H.%M.%S");
296        let backup_path = path.with_extension(format!("bak.{timestamp}.json"));
297        tracing::debug!("Creating backup at: {:?}", backup_path);
298        let _ = std::fs::copy(path, &backup_path);
299
300        if let Some(parent) = path.parent() {
301            let prefix = path.file_stem().unwrap_or_default().to_string_lossy();
302            let mut backups: Vec<_> = std::fs::read_dir(parent)
303                .unwrap_or_else(|_| std::fs::read_dir(".").unwrap())
304                .filter_map(|entry| entry.ok())
305                .filter(|entry| {
306                    entry
307                        .file_name()
308                        .to_string_lossy()
309                        .starts_with(&format!("{prefix}.bak."))
310                        && entry.file_name().to_string_lossy().ends_with(".json")
311                })
312                .collect();
313
314            if backups.len() > 5 {
315                backups.sort_by_key(|entry| {
316                    entry
317                        .metadata()
318                        .and_then(|m| m.modified())
319                        .unwrap_or(std::time::UNIX_EPOCH)
320                });
321                for old_backup in backups.iter().take(backups.len() - 5) {
322                    let _ = std::fs::remove_file(old_backup.path());
323                }
324            }
325        }
326    }
327
328    tracing::debug!("Saving config to: {:?}", path);
329    save_config(path, &config);
330
331    tracing::debug!("Config loading completed");
332    config
333}
