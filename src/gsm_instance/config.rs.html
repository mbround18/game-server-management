1//! # Instance Configuration
2//!
3//! This module defines the structures and enumerations used to configure a game server instance.
4//! The central piece is the `InstanceConfig` struct, which holds all the necessary settings
5//! for installing, running, and managing a game server.
6use serde::{Deserialize, Serialize};
7use std::path::PathBuf;
8
9/// Defines the launch mode for the game server.
10///
11/// This enum allows specifying how the game server executable should be run, which is
12/// particularly useful for handling cross-platform compatibility (e.g., running a
13/// Windows-based server on Linux).
14#[derive(Debug, Clone, Serialize, Deserialize)]
15pub enum LaunchMode {
16    /// Run the server executable natively. This is the default.
17    Native,
18    /// Run the server using Wine. This is typically used for Windows executables on Linux.
19    Wine,
20    /// Run the server using Proton, which is Valve's compatibility tool for running
21    /// Windows games on Linux.
22    Proton,
23}
24
25/// Configuration for a game server instance managed by `gsm-instance`.
26///
27/// This struct holds all the parameters needed to configure and manage a game server,
28/// from installation with SteamCMD to launching the server process. It is the primary
29/// configuration object used throughout the `gsm-instance` crate.
30///
31/// # Example
32///
33/// ```rust
34/// use gsm_instance::config::{InstanceConfig, LaunchMode};
35/// use std::path::PathBuf;
36///
37/// let config = InstanceConfig {
38///     app_id: 123456,
39///     name: "My Awesome Server".to_string(),
40///     command: "server_executable".to_string(),
41///     install_args: vec!["+beta".to_string(), "preview".to_string()],
42///     launch_args: vec!["-nographics".to_string(), "-batchmode".to_string()],
43///     force_windows: true,
44///     working_dir: PathBuf::from("/home/steam/myserver"),
45///     launch_mode: LaunchMode::Proton,
46/// };
47/// ```
48#[derive(Debug, Clone, Serialize, Deserialize)]
49pub struct InstanceConfig {
50    /// The Steam App ID for the game server. This is used by SteamCMD to identify which
51    /// game server to install or update.
52    pub app_id: u32,
53    /// A user-friendly name for the server, primarily used for logging and identification.
54    pub name: String,
55    /// The command or executable path used to launch the server. This can be a simple
56    /// executable name (e.g., `valheim_server.x86_64`) or a path relative to the
57    /// `working_dir`.
58    pub command: String,
59    /// A list of additional arguments to pass to SteamCMD during the installation or
60    /// update process. This can be used for things like selecting a beta branch.
61    pub install_args: Vec<String>,
62    /// A list of additional arguments to pass to the server executable when it is launched.
63    pub launch_args: Vec<String>,
64    /// If `true`, forces the installation and launch of the Windows version of the game
65    /// server, typically for use with Wine or Proton on Linux.
66    pub force_windows: bool,
67    /// The working directory where the server will be installed and run. All server-related
68    /// files, logs, and the PID file will be stored here.
69    pub working_dir: PathBuf,
70    /// The launch mode for the server, which determines how the executable is run.
71    pub launch_mode: LaunchMode,
72}
73
74impl Default for InstanceConfig {
75    /// Creates a default `InstanceConfig` with empty or default values.
76    ///
77    /// The default configuration is not typically useful on its own, but it provides a
78    /// convenient starting point for building a configuration.
79    fn default() -> Self {
80        Self {
81            app_id: 0,
82            name: String::new(),
83            command: String::new(),
84            install_args: Vec::new(),
85            launch_args: Vec::new(),
86            force_windows: false,
87            working_dir: std::env::current_dir().unwrap_or_else(|_| PathBuf::from(".")),
88            launch_mode: LaunchMode::Native,
89        }
90    }
91}
92
93impl InstanceConfig {
94    /// Returns the path to the PID file for the instance.
95    ///
96    /// The PID file is used to store the process ID of the running server, which allows
97    /// for managing the server process (e.g., stopping or checking its status).
98    pub fn pid_file(&self) -> PathBuf {
99        self.working_dir.join("instance.pid")
100    }
101
102    /// Returns the path to the log directory for the instance.
103    pub fn log_dir(&self) -> PathBuf {
104        self.working_dir.join("logs")
105    }
106
107    /// Returns the path to the standard output log file for the server.
108    pub fn stdout(&self) -> PathBuf {
109        self.log_dir().join("server.log")
110    }
111
112    /// Returns the path to the standard error log file for the server.
113    pub fn stderr(&self) -> PathBuf {
114        self.log_dir().join("server.err")
115    }
116}

118#[cfg(test)]
119mod tests {
120    use super::{InstanceConfig, LaunchMode};
121
122    #[test]
123    fn default_config_uses_empty_values_and_native_mode() {
124        let config = InstanceConfig::default();
125
126        assert_eq!(config.app_id, 0);
127        assert_eq!(config.name, "");
128        assert_eq!(config.command, "");
129        assert!(config.install_args.is_empty());
130        assert!(config.launch_args.is_empty());
131        assert!(!config.force_windows);
132        assert!(matches!(config.launch_mode, LaunchMode::Native));
133    }
134
135    #[test]
136    fn path_helpers_are_relative_to_working_dir() {
137        let working_dir = std::env::temp_dir().join("gsm-instance-config-tests");
138        let config = InstanceConfig {
139            working_dir: working_dir.clone(),
140            ..InstanceConfig::default()
141        };
142
143        assert_eq!(config.pid_file(), working_dir.join("instance.pid"));
144        assert_eq!(config.log_dir(), working_dir.join("logs"));
145        assert_eq!(config.stdout(), working_dir.join("logs").join("server.log"));
146        assert_eq!(config.stderr(), working_dir.join("logs").join("server.err"));
147    }
148
149    #[test]
150    fn serde_round_trip_preserves_non_default_fields() {
151        let config = InstanceConfig {
152            app_id: 2278520,
153            name: String::from("Test Server"),
154            command: String::from("./server"),
155            install_args: vec![String::from("+beta"), String::from("staging")],
156            launch_args: vec![String::from("-log"), String::from("-port=27015")],
157            force_windows: true,
158            working_dir: std::path::PathBuf::from("/srv/server"),
159            launch_mode: LaunchMode::Proton,
160        };
161
162        let serialized = serde_json::to_string(&config).expect("serialize config");
163        let deserialized: InstanceConfig =
164            serde_json::from_str(&serialized).expect("deserialize config");
165
166        assert_eq!(deserialized.app_id, 2278520);
167        assert_eq!(deserialized.name, "Test Server");
168        assert_eq!(deserialized.command, "./server");
169        assert_eq!(deserialized.install_args, vec!["+beta", "staging"]);
170        assert_eq!(deserialized.launch_args, vec!["-log", "-port=27015"]);
171        assert!(deserialized.force_windows);
172        assert_eq!(
173            deserialized.working_dir,
174            std::path::PathBuf::from("/srv/server")
175        );
176        assert!(matches!(deserialized.launch_mode, LaunchMode::Proton));
177    }
