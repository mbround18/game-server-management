1//! # Game Server Launcher
2//!
3//! This module is responsible for launching the game server executable. It handles different
4//! launch modes, including native execution, and compatibility layers like Wine and Proton
5//! for running Windows servers on Linux.
6//!
7//! The main entry point is the `launch_server` function, which takes an `InstanceConfig`
8//! and constructs a `Command` ready to be executed.
9use crate::config::InstanceConfig;

13use crate::proton::ProtonConfig;
14use std::env;

18use std::process::{Command, Stdio};

22/// Represents the Windows compatibility layer to use for launching the server.
23enum WindowsCompat {
24    /// Use Proton, with a specific `ProtonConfig`.
25    Proton { config: ProtonConfig },
26    /// Use Wine, with the specified path to the Wine executable.
27    Wine { path: String },
28    /// No compatibility layer.
29    None,
30}
31
32impl WindowsCompat {
33    /// Creates a `Command` for the given game executable using the compatibility layer.
34    fn create_command(&self, game_exe: &str) -> Option<Command> {
35        match self {
36            WindowsCompat::Proton { config } => {
37                debug!("Creating Proton command for: {}", game_exe);
38                Some(config.create_command(game_exe))
39            }
40            WindowsCompat::Wine { path } => {
41                debug!("Creating Wine command for: {}", game_exe);
42                let mut cmd = Command::new(path);
43                cmd.arg(game_exe);
44                Some(cmd)
45            }
46            WindowsCompat::None => {
47                debug!("No compatibility layer available for: {}", game_exe);
48                None
49            }
50        }
51    }
52}

54/// Tries to find a Proton installation, either a specific version or any available version.
55fn try_find_proton(
56    version_option: Option<&str>,
57    force_proton: bool,
58    app_id: u32,
59) -> Result<WindowsCompat, String> {
60    match proton::find_proton(version_option) {
61        Ok(mut config) => {
62            let version_desc = version_option.unwrap_or("any version");
63            debug!("Found Proton {} at {}", version_desc, config.path);
64            config.app_id = app_id.to_string();
65            setup_proton_config(config)
66        }
67        Err(e) => {
68            let err_msg = match version_option {
69                Some(v) => format!("Failed to find or download Proton {}: {}", v, e),
70                None => format!("Proton not found: {}", e),
71            };
72
73            if version_option.is_some() {
74                error!("{}", err_msg);
75            } else {
76                debug!("{}", err_msg);
77            }
78
79            if force_proton {
80                Err(err_msg)
81            } else {
82                Err(format!("Proton unavailable: {}", e))
83            }
84        }
85    }
86}

88/// Sets up the Proton prefix and environment variables for a given `ProtonConfig`.
89fn setup_proton_config(mut config: ProtonConfig) -> Result<WindowsCompat, String> {
90    if let Ok(home) = env::var("HOME") {
91        let prefix_path = format!("{}/.proton/prefixes/gsm", home);
92        debug!("Setting up Proton prefix at: {}", prefix_path);
93        if let Err(e) = proton::setup_prefix(&mut config, &prefix_path) {
94            error!("Failed to set up Proton prefix: {}", e);
95        } else {
96            debug!("Successfully set up Proton prefix");
97        }
98    }
99
100    debug!("Initializing Proton environment variables");
101    if let Err(e) = proton::init_proton_env(&mut config) {
102        error!("Failed to initialize Proton environment: {}", e);
103    } else {
104        debug!("Successfully initialized Proton environment");
105    }
106
107    Ok(WindowsCompat::Proton { config })
108}

110/// Checks if a string value represents a truthy value.
111fn is_truthy(val: &str) -> bool {
112    val == "1" || val == "true" || val == "yes"
113}

115/// Finds a suitable Windows compatibility layer (Proton or Wine) based on the launch mode
116/// and environment variables.
117fn find_windows_compatibility(
118    app_id: u32,
119    launch_mode: &LaunchMode,
120) -> Result<WindowsCompat, String> {
121    debug!("Searching for Windows compatibility layers");
122    let force_proton = env::var("FORCE_PROTON")
123        .map(|v| is_truthy(&v))
124        .unwrap_or(false);
125
126    if let LaunchMode::Proton = launch_mode {
127        // Check if PROTON_VERSION is set
128        if let Ok(version) = env::var("PROTON_VERSION") {
129            debug!("PROTON_VERSION is set to: {}", version);
130
131            let parsed_version = match proton::parse_version(&version) {
132                Ok(v) => v,
133                Err(e) => {
134                    error!("Failed to parse PROTON_VERSION: {}", e);
135                    version
136                }
137            };
138
139            let result = try_find_proton(Some(&parsed_version), force_proton, app_id);
140            if result.is_ok() || force_proton {
141                return result;
142            }
143        }
144
145        // If no specific version requested, try to find any version
146        let result = try_find_proton(None, force_proton, app_id);
147        if result.is_ok() || force_proton {
148            return result;
149        }
150    }
151
152    if let LaunchMode::Wine = launch_mode {
153        if let Ok(wine_path) = find_wine() {
154            debug!("Found Wine at: {}", wine_path);
155            return Ok(WindowsCompat::Wine { path: wine_path });
156        } else {
157            debug!("Wine not found");
158        }
159    }
160
161    if force_proton {
162        return Err("FORCE_PROTON is set but no Proton installation was found".to_string());
163    }
164
165    debug!("No Windows compatibility layer found");
166    Ok(WindowsCompat::None)
167}

169/// Finds the path to the Wine executable (`wine64` or `wine`).
170fn find_wine() -> Result<String, String> {
171    // Attempt to find 'wine64' first
172    if let Ok(path) = which("wine64") {
173        return path
174            .to_str()
175            .map(|s| s.to_string())
176            .ok_or_else(|| "Failed to convert wine64 path to string.".to_string());
177    }
178    // If 'wine64' is not found, attempt to find 'wine'
179    if let Ok(path) = which("wine") {
180        return path
181            .to_str()
182            .map(|s| s.to_string())
183            .ok_or_else(|| "Failed to convert wine path to string.".to_string());
184    }
185    // If neither is found, return an error
186    Err("Neither 'wine64' nor 'wine' was found in the system's PATH.".to_string())
187}

189/// Creates a `Command` for a Windows executable, using a compatibility layer if available.
190fn get_command_for_windows(
191    exe_path: &str,
192    app_id: u32,
193    launch_mode: &LaunchMode,
194) -> Result<Command, InstanceError> {
195    debug!("Getting Windows command for: {}", exe_path);
196
197    // Try to find a suitable Windows compatibility layer
198    let compat = find_windows_compatibility(app_id, launch_mode).map_err(|e| {
199        // Check if we need to exit immediately due to FORCE_PROTON
200        if env::var("FORCE_PROTON")
201            .map(|v| is_truthy(&v))
202            .unwrap_or(false)
203        {
204            error!("FORCE_PROTON set but Proton setup failed: {}", e);
205            std::process::exit(1);
206        }
207        error!("Failed to find Windows compatibility layer: {}", e);
208        InstanceError::CommandExecutionError(e)
209    })?;
210
211    match &compat {
212        WindowsCompat::Proton { config } => {
213            let cmd_exists = Path::new(&config.path).exists();
214            debug!("Using Proton at: {} (exists: {})", config.path, cmd_exists);
215            if !cmd_exists {
216                error!("Proton executable not found at: {}", config.path);
217            }
218        }
219        WindowsCompat::Wine { path } => {
220            let cmd_exists = Path::new(path).exists();
221            debug!("Using Wine at: {} (exists: {})", path, cmd_exists);
222            if !cmd_exists {
223                error!("Wine executable not found at: {}", path);
224            }
225        }
226        WindowsCompat::None => debug!("No Windows compatibility layer selected"),
227    }
228
229    let cmd = compat.create_command(exe_path).ok_or_else(|| {
230        error!("Failed to create command for {}", exe_path);
231        InstanceError::CommandExecutionError(
232            "No suitable Windows compatibility layer found.".to_string(),
233        )
234    })?;
235
236    debug!("Created Windows command: {:?}", cmd);
237    Ok(cmd)
238}

240/// Prepares a `Command` to launch the game server based on the provided configuration.
241///
242/// This function constructs a `Command` that is ready to be spawned as a child process.
243/// It sets up the executable, arguments, working directory, and log files based on the
244/// `InstanceConfig`.
245///
246/// # Arguments
247///
248/// * `config`: A reference to the `InstanceConfig` for the server.
249///
250/// # Returns
251///
252/// A `Result` containing the configured `Command`, or an `InstanceError` if something
253/// goes wrong (e.g., a log file cannot be created).
254///
255/// # Behavior
256///
257/// - If `launch_mode` is `Native`, it creates a simple command for the executable.
258/// - If `launch_mode` is `Proton` or `Wine`, it attempts to find a suitable compatibility
259///   layer and constructs the command accordingly.
260/// - It appends any `launch_args` from the configuration.
261/// - It sets the working directory to `config.working_dir`.
262/// - It creates the log directory and redirects the command's `stdout` and `stderr` to
263///   log files (`server.log` and `server.err`).
264pub fn launch_server(config: &InstanceConfig) -> Result<Command, InstanceError> {
265    debug!("Launching server with config: {:?}", config);
266
267    let mut command = match config.launch_mode {
268        LaunchMode::Native => {
269            debug!("Using native command: {}", config.command);
270            Command::new(&config.command)
271        }
272        LaunchMode::Proton | LaunchMode::Wine => {
273            debug!("Windows executable detected, finding compatibility layer");
274            get_command_for_windows(&config.command, config.app_id, &config.launch_mode)?
275        }
276    };
277
278    // Append additional launch arguments.
279    if !config.launch_args.is_empty() {
280        debug!("Adding launch args: {:?}", config.launch_args);
281        for arg in &config.launch_args {
282            command.arg(arg);
283        }
284    }
285
286    // Set the working directory.
287    debug!("Setting working directory: {:?}", config.working_dir);
288    command.current_dir(&config.working_dir);
289
290    if let Err(e) = create_dir_all(config.log_dir()) {
291        error!("Failed to create log directory: {}", e);
292        return Err(InstanceError::IoError(e));
293    }
294
295    debug!("Creating stdout log file at: {:?}", config.stdout());
296    let stdout_file = match File::create(config.stdout()) {
297        Ok(file) => file,
298        Err(e) => {
299            error!("Failed to create stdout log file: {}", e);
300            return Err(InstanceError::IoError(e));
301        }
302    };
303
304    debug!("Creating stderr log file at: {:?}", config.stderr());
305    let stderr_file = match File::create(config.stderr()) {
306        Ok(file) => file,
307        Err(e) => {
308            error!("Failed to create stderr log file: {}", e);
309            return Err(InstanceError::IoError(e));
310        }
311    };
312
313    command.stdout(Stdio::from(stdout_file));
314    command.stderr(Stdio::from(stderr_file));
315
316    debug!("Final command: {:?}", command);
317
318    Ok(command)
319}
