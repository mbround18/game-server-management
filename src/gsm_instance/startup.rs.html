1//! # Game Server Startup
2//!
3//! This module provides functionality for starting a game server process,
4//! particularly focusing on daemonizing the process and managing its lifecycle.
5//! It handles the creation of necessary directories and the redirection of
6//! standard output/error to log files.
7use crate::config::InstanceConfig;
8use crate::errors::InstanceError;
9use crate::launcher::launch_server;
10use std::fs;
11use std::fs::create_dir_all;
12use std::path::Path;
13use std::process::Child;
14use std::thread;
15use std::time::Duration;
16use tracing::info;

18/// Ensures the log directory exists under the given working directory.
19///
20/// This helper function creates the `logs` subdirectory within the specified
21/// `working_dir` if it doesn't already exist. This is where the server's
22/// standard output and error streams will be redirected.
23///
24/// # Arguments
25///
26/// * `working_dir`: The root working directory of the game server instance.
27///
28/// # Returns
29///
30/// Returns `Ok(())` on success, or an `InstanceError::IoError` if the directory
31/// cannot be created.
32fn ensure_log_dir(working_dir: &Path) -> Result<(), InstanceError> {
33    let logs_dir = working_dir.join("logs");
34    create_dir_all(&logs_dir)?;
35    Ok(())
36}

38/// Starts the game server as a daemonized process.
39///
40/// This function orchestrates the launch of the game server. It first prepares
41/// the logging environment, then uses the `launcher` module to construct the
42/// appropriate command. The server process is then spawned, its PID is recorded
43/// in a file, and the function waits briefly to catch immediate startup failures.
44///
45/// # Arguments
46///
47/// * `config`: The `InstanceConfig` containing all necessary settings for the server.
48///
49/// # Returns
50///
51/// Returns a `Result` containing a `Child` handle to the spawned server process
52/// on success, or an `InstanceError` if the startup fails at any stage.
53///
54/// # Behavior
55///
56/// - Creates a `logs` directory within the `working_dir` if it doesn't exist.
57/// - Constructs the launch command using `launcher::launch_server`.
58/// - Spawns the server process in the background.
59/// - Writes the process ID (PID) of the spawned server to an `instance.pid` file
60///   within the `working_dir`. This PID file is crucial for managing the server's
61///   lifecycle (e.g., stopping it).
62/// - Waits for a short duration (10 seconds) after spawning to detect if the server
63///   process immediately exits, indicating a startup failure. If it exits, the PID
64///   file is removed, and an error is returned.
65/// - Standard output and error of the child process are redirected to `server.log`
66///   and `server.err` files in the `logs` directory.
67pub fn start_daemonized(config: InstanceConfig) -> Result<Child, InstanceError> {
68    info!("Starting server as a daemonized process...");
69    let working_dir = config.working_dir.clone();
70    ensure_log_dir(&working_dir)?;
71
72    match launch_server(&config) {
73        Ok(mut cmd) => match cmd.spawn() {
74            Ok(mut child) => {
75                let pid = child.id();
76                let pid_file = working_dir.join("instance.pid");
77
78                if pid_file.exists() {
79                    fs::remove_file(&pid_file)?;
80                }
81
82                fs::write(pid_file, pid.to_string())?;
83
84                // Surface immediate startup failures so callers do not assume
85                // a zombie/failed process is a healthy server start.
86                // Some proton/wine launch failures occur a few seconds after
87                // process creation; wait briefly to catch those as start errors.
88                thread::sleep(Duration::from_secs(10));
89                if let Some(status) = child
90                    .try_wait()
91                    .map_err(|e| InstanceError::CommandExecutionError(e.to_string()))?
92                {
93                    let _ = fs::remove_file(working_dir.join("instance.pid"));
94                    return Err(InstanceError::CommandExecutionError(format!(
95                        "Server process exited immediately with status {status}"
96                    )));
97                }
98
99                Ok(child)
100            }
101            Err(e) => Err(InstanceError::CommandExecutionError(e.to_string())),
102        },
103        Err(e) => Err(InstanceError::CommandExecutionError(e.to_string())),
104    }
105}
