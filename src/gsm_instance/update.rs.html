1//! # Update Module
2//!
3//! This module provides functionality to check for and perform updates of the game server.
4//!
5//! It compares the build IDs from the current app manifest and the latest app info from SteamCMD.
6//! If an update is available (i.e. the build IDs differ), the `update_server` function can be used
7//! to update the installation via SteamCMD.
8//!
9//! ## Example
10//!
11//! ```rust,no_run
12//! use std::path::Path;
13//! use gsm_instance::update::{update_is_available, update_server};
14//! use gsm_instance::errors::InstanceError;
15//!
16//! // Paths to the manifest and app info files
17//! let manifest_path = Path::new("/home/steam/myserver/steamapps/appmanifest_123456.acf");
18//! let appinfo_path = Path::new("/home/steam/Steam/appcache/appinfo.vdf");
19//!
20//! // Check if an update is available
21//! let available = update_is_available(manifest_path, appinfo_path)?;
22//! if available {
23//!     // Run the update with any extra arguments (if needed)
24//!     update_server(123456, Path::new("/home/steam/myserver"), false, &vec!["verbose".to_string()])?;
25//! }
26//! # Ok::<(), InstanceError>(())
27//! ```
28
29use crate::errors::InstanceError;
30use crate::steamcmd::steamcmd_command;
31use regex::Regex;
32use std::fs;
33use std::path::Path;
34use tracing::{debug, info};

36/// Struct holding build ID information.
37#[derive(Debug, PartialEq, Eq)]
38pub struct UpdateInfo {
39    pub current_build_id: String,
40    pub latest_build_id: String,
41}
42
43impl UpdateInfo {
44    /// Creates a new UpdateInfo by reading the manifest and appinfo files.
45    pub fn new(manifest_path: &Path, appinfo_path: &Path) -> Result<Self, InstanceError> {
46        let manifest_data = fs::read_to_string(manifest_path)
47            .map_err(|e| InstanceError::CommandExecutionError(e.to_string()))?;
48        let current_build_id = extract_build_id_from_manifest(&manifest_data).to_string();
49
50        let appinfo_data = fs::read_to_string(appinfo_path)
51            .map_err(|e| InstanceError::CommandExecutionError(e.to_string()))?;
52        let latest_build_id = extract_build_id_from_app_info(&appinfo_data).to_string();
53
54        Ok(UpdateInfo {
55            current_build_id,
56            latest_build_id,
57        })
58    }
59
60    /// Returns true if an update is available (build IDs differ).
61    pub fn update_available(&self) -> bool {
62        self.current_build_id != self.latest_build_id
63    }
64}
65
66/// Extracts the build ID from the manifest file contents using regex.
67///
68/// Expected format: `"buildid"    "123456"`.
69fn extract_build_id_from_manifest(manifest: &str) -> &str {
70    let re = Regex::new(r#""buildid"\s+"(\d+)""#).unwrap();
71    if let Some(caps) = re.captures(manifest) {
72        caps.get(1).map_or("", |m| m.as_str())
73    } else {
74        panic!("Failed to extract buildid from manifest:\n{manifest}");
75    }
76}
77
78/// Extracts the build ID from the appinfo file contents using regex.
79///
80/// Expected format (simplified): `"buildid"    "123456"`.
81fn extract_build_id_from_app_info(app_info: &str) -> &str {
82    let re = Regex::new(r#""buildid"\s+"(\d+)""#).unwrap();
83    if let Some(caps) = re.captures(app_info) {
84        caps.get(1).map_or("", |m| m.as_str())
85    } else {
86        panic!("Failed to extract buildid from appinfo:\n{app_info}");
87    }
88}

90/// Checks if an update is available by comparing the build IDs from the manifest and appinfo files.
91pub fn update_is_available(
92    manifest_path: &Path,
93    appinfo_path: &Path,
94) -> Result<bool, InstanceError> {
95    let update_info = UpdateInfo::new(manifest_path, appinfo_path)?;
96    debug!("Update info: {:?}", update_info);
97    Ok(update_info.update_available())
98}

100/// Updates the server installation using SteamCMD.
101///
102/// # Parameters
103/// - `app_id`: The Steam App ID of the server.
104/// - `install_dir`: The directory where the server is installed.
105/// - `extra_args`: Additional arguments to pass to SteamCMD during update.
106///
107/// # Behavior
108/// Builds a SteamCMD command to update the app (with validation) and executes it.
109/// Returns an error if the command fails.
110///
111/// # Example
112/// ```rust,no_run
113/// # use std::path::Path;
114/// # use gsm_instance::update::update_server;
115/// update_server(123456, Path::new("/home/steam/myserver"), false, &vec!["verbose".to_string()]).expect("Update failed");
116/// ```
117pub fn update_server<P: AsRef<Path>>(
118    app_id: u32,
119    install_dir: P,
120    force_windows: bool,
121    extra_args: &[String],
122) -> Result<(), InstanceError> {
123    info!(
124        "Updating app {} in {}",
125        app_id,
126        install_dir.as_ref().display()
127    );
128    let login = "+login anonymous".to_string();
129    let force_install_dir = format!("+force_install_dir {}", install_dir.as_ref().display());
130    let app_update = format!("+app_update {app_id} validate");
131    let mut args = vec![force_install_dir, login, app_update];
132
133    if force_windows {
134        let platform = "windows";
135        args.insert(0, format!("+@sSteamCmdForcePlatformType {platform}"));
136    }
137
138    args.extend_from_slice(extra_args);
139    args.push(String::from("+quit"));
140
141    let mut steamcmd = steamcmd_command();
142    let command = steamcmd.args(&args);
143    debug!("Executing update command: {:?}", command);
144    let output = command
145        .output()
146        .map_err(|e| InstanceError::CommandExecutionError(e.to_string()))?;
147    if output.status.success() {
148        info!("Update successful.");
149        Ok(())
150    } else {
151        Err(InstanceError::CommandExecutionError(format!(
152            "Update failed with status: {:?}",
153            output.status
154        )))
155    }
156}

180    #[cfg(unix)]
181    fn write_executable_script(path: &Path, body: &str) {
182        use std::os::unix::fs::PermissionsExt;
183
184        fs::write(path, body).unwrap();
185        let mut permissions = fs::metadata(path).unwrap().permissions();
186        permissions.set_mode(0o755);
187        fs::set_permissions(path, permissions).unwrap();
188    }

259        let _lock = env_lock().lock().unwrap_or_else(|e| e.into_inner());
260        let temp_dir = tempdir().unwrap();

282        );
283        assert_eq!(lines[2], "+login anonymous");
284        assert_eq!(lines[3], "+app_update 2278520 validate");
285        assert_eq!(lines[4], "+app_info_update 1");
286        assert_eq!(lines[5], "+quit");

288        unsafe {
289            std::env::remove_var("STEAMCMD_PATH");
290        }
