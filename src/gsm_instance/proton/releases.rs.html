1//! # Proton Releases
2//!
3//! This module provides functionality for fetching information about Proton GE releases
4//! from the GitHub API. It allows listing all available releases, fetching the latest
5//! release, or fetching a specific release by version tag.
6use super::types::ProtonRelease;
7use reqwest;
8use serde::Deserialize;
9use thiserror::Error;
10
11/// Represents errors that can occur while fetching Proton release information.
12#[derive(Error, Debug)]
13pub enum ReleaseError {
14    /// A network error occurred while making a request to the GitHub API.
15    #[error("Network error: {0}")]
16    Network(#[from] reqwest::Error),
17
18    /// An error occurred while parsing the JSON response from the GitHub API.
19    #[error("Failed to parse JSON: {0}")]
20    Json(#[from] serde_json::Error),
21
22    /// The requested release was not found.
23    #[error("Release not found: {0}")]
24    NotFound(String),
25}
26
27/// Represents a release from the GitHub API.
28#[derive(Deserialize, Debug)]
29struct GitHubRelease {
30    tag_name: String,
31    published_at: String,
32    assets: Vec<GitHubAsset>,
33}
34
35/// Represents an asset within a GitHub release.
36#[derive(Deserialize, Debug)]
37struct GitHubAsset {
38    browser_download_url: String,
39    name: String,
40}
41
42const GITHUB_API_URL: &str =
43    "https://api.github.com/repos/GloriousEggroll/proton-ge-custom/releases";
44
45/// Fetches a list of all available Proton GE releases from GitHub.
46///
47/// This function makes a request to the GitHub API to get a list of all releases
48/// for the `proton-ge-custom` repository. It then filters these releases to find
49/// the ones that have a `.tar.gz` asset, which is the format used for distribution.
50pub fn list_available_releases() -> Result<Vec<ProtonRelease>, ReleaseError> {
51    let releases: Vec<GitHubRelease> = reqwest::blocking::Client::new()
52        .get(GITHUB_API_URL)
53        .header("User-Agent", "gsm-instance")
54        .send()?
55        .json()?;
56
57    let proton_releases = releases
58        .into_iter()
59        .filter_map(|r| {
60            r.assets
61                .iter()
62                .find(|&a| a.name.ends_with(".tar.gz"))
63                .map(|a| ProtonRelease {
64                    tag: r.tag_name.clone(),
65                    download_url: a.browser_download_url.clone(),
66                    release_date: r.published_at.clone(),
67                })
68        })
69        .collect();
70
71    Ok(proton_releases)
72}
73
74/// Fetches the latest available Proton GE release from GitHub.
75pub fn fetch_latest_release() -> Result<ProtonRelease, ReleaseError> {
76    let releases = list_available_releases()?;
77    releases
78        .into_iter()
79        .next()
80        .ok_or_else(|| ReleaseError::NotFound("No releases found".to_string()))
81}
82
83/// Fetches a specific Proton GE release by its version tag.
84///
85/// # Arguments
86///
87/// * `version`: The tag of the release to fetch (e.g., "GE-Proton8-25").
88pub fn fetch_specific_release(version: &str) -> Result<ProtonRelease, ReleaseError> {
89    let releases = list_available_releases()?;
90    releases
91        .into_iter()
92        .find(|r| r.tag == version)
93        .ok_or_else(|| ReleaseError::NotFound(format!("Release {} not found", version)))
94}
